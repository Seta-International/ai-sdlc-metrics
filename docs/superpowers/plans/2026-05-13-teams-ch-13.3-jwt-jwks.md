# Teams JWT/JWKS Verifier — Implementation Plan (WBS 13.3)

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Verify every inbound `POST /teams/messages` against a real Bot Framework JWT. Remove `TEAMS_SKIP_JWT_VERIFY` entirely — dev and prod behave identically after this task.

**Architecture:** A new `jwt.ts` module holds JWKS verification state (module-level singleton). The JWKS cache is persisted to a new `platform.jwks_cache` Postgres table so restarts don't hammer the Bot Framework JWKS endpoint. `verifyBotFrameworkJwt(token, botId, sql)` accepts `sql` as a parameter so the cache access is testable and follows the workspace's injection pattern (`@seta/db` provides pool/connection, not a Drizzle instance). Three critical `jose` 6.x patterns must be followed exactly (see Background below).

**Tech Stack:** jose 6.x, Drizzle ORM, msw 2.7.x, vitest.

**Spec:** `docs/superpowers/specs/2026-05-13-teams-ch-13.3-jwt-jwks-design.md`

**Depends on:** WBS 13.2 merged.

---

## Background: three required jose 6.x patterns

Deviating from any of these causes silent failures or security holes:

1. `createRemoteJWKSet` takes a **URL object** (`new URL(...)`), not a string. Passing a string silently breaks key loading in jose 6.x.
2. `jwtVerify` must pin `algorithms: ['RS256']`. Omitting this allows "none" and HS-confusion attacks.
3. During MS key rotation, two valid keys briefly share the same `kid`. `jose` throws `ERR_JWKS_MULTIPLE_MATCHING_KEYS` in this case. Catch it and call `jwtVerify` a second time — jose tries the next matching key on the second call.

### Task 6: Create `jwt.ts`

**Files:** Create `modules/channels/teams/src/jwt.ts`

- [ ] **Write the file** — read carefully; all three jose patterns from the Background section are implemented here

```ts
import * as jose from 'jose'
import type { DbSql } from '@seta/db'
import { BotFrameworkJwtInvalid } from './errors.js'
import { logger } from '@seta/observability'

const BOT_FRAMEWORK_JWKS_URL = new URL('https://login.botframework.com/v1/.well-known/keys')
const BOT_FRAMEWORK_ISSUER   = 'https://api.botframework.com'
const CACHE_KEY               = 'botframework'

// Module-level singletons — one JWKS instance per process lifetime
let JWKS: ReturnType<typeof jose.createRemoteJWKSet> | undefined
let jwksCache: jose.JWKSCacheInput = {}

async function initJwks(sql: DbSql): Promise<ReturnType<typeof jose.createRemoteJWKSet>> {
  if (JWKS) return JWKS
  const persisted = await readJwksCacheFromDb(sql)
  jwksCache = persisted ?? {}
  // Pattern 1: URL object — NOT a string
  JWKS = jose.createRemoteJWKSet(BOT_FRAMEWORK_JWKS_URL, {
    cooldownDuration: 30_000,
    [jose.jwksCache]: jwksCache,
  })
  return JWKS
}

export async function verifyBotFrameworkJwt(
  token: string,
  botId: string,
  sql: DbSql,
): Promise<jose.JWTPayload> {
  const jwks = await initJwks(sql)
  const prevUat = jwksCache.uat

  // Pattern 2: pin algorithms — never allow 'none' or HS variants
  const verifyOpts: jose.JWTVerifyOptions = {
    issuer:         BOT_FRAMEWORK_ISSUER,
    audience:       botId,
    algorithms:     ['RS256'],
    clockTolerance: 60,   // Bot Framework clocks drift up to ~60 s
  }

  try {
    const { payload } = await jose.jwtVerify(token, jwks, verifyOpts)
    if (jwksCache.uat !== prevUat) await writeJwksCacheToDb(sql, jwksCache)
    return payload
  } catch (err) {
    // Pattern 3: key rotation — retry once on multi-match
    if (
      err instanceof Error &&
      'code' in err &&
      (err as { code: string }).code === 'ERR_JWKS_MULTIPLE_MATCHING_KEYS'
    ) {
      logger.warn('JWKS multi-match during key rotation — retrying')
      const { payload } = await jose.jwtVerify(token, jwks, verifyOpts)
      if (jwksCache.uat !== prevUat) await writeJwksCacheToDb(sql, jwksCache)
      return payload
    }
    throw new BotFrameworkJwtInvalid(err instanceof Error ? err.message : String(err))
  }
}

async function readJwksCacheFromDb(sql: DbSql): Promise<jose.JWKSCacheInput | null> {
  const rows = await sql<Array<{ payload: jose.JWKSCacheInput }>>`
    SELECT payload FROM platform.jwks_cache WHERE key = ${CACHE_KEY} LIMIT 1
  `
  return rows[0]?.payload ?? null
}

async function writeJwksCacheToDb(sql: DbSql, cache: jose.JWKSCacheInput): Promise<void> {
  await sql`
    INSERT INTO platform.jwks_cache (key, payload, updated_at)
    VALUES (${CACHE_KEY}, ${sql.json(cache as object)}, NOW())
    ON CONFLICT (key) DO UPDATE
      SET payload     = excluded.payload,
          updated_at  = NOW()
  `
}
```

> **Note on `sql.json()`:** The `DbSql` type from `@seta/db` is a postgres.js tagged-template pool. Use `sql.json(value)` for JSONB parameters — passing a plain object directly causes a type error with postgres.js.

```ts
import * as jose from 'jose'
import { http, HttpResponse } from 'msw'
import { setupServer } from 'msw/node'
import { afterAll, afterEach, beforeAll, beforeEach, describe, expect, test, vi } from 'vitest'
import { BotFrameworkJwtInvalid } from './errors.js'
import { verifyBotFrameworkJwt } from './jwt.js'
import type { DbSql } from '@seta/db'

// Generate a test RSA key pair once for all tests
const { privateKey, publicKey } = await jose.generateKeyPair('RS256')
const BOT_ID = 'test-bot-id'

async function signTestJwt(overrides: Record<string, unknown> = {}): Promise<string> {
  return new jose.SignJWT({ aud: BOT_ID, ...overrides })
    .setProtectedHeader({ alg: 'RS256', kid: 'test-key-1' })
    .setIssuer('https://api.botframework.com')
    .setIssuedAt()
    .setExpirationTime('1h')
    .sign(privateKey)
}

async function buildJwks(): Promise<jose.JSONWebKeySet> {
  const pub = await jose.exportJWK(publicKey)
  return { keys: [{ ...pub, kid: 'test-key-1', use: 'sig' }] }
}

// Minimal sql stub — cache tests inject real SQL; JWT tests only need a no-cache stub
function makeSqlStub(cacheRow?: jose.JWKSCacheInput): DbSql {
  const stub = vi.fn() as unknown as DbSql & { json: (v: object) => object }
  // Template tag: return empty array (no cached row) by default
  const fn = Object.assign(
    (strings: TemplateStringsArray, ...values: unknown[]) => {
      const query = strings.join('?')
      if (query.includes('SELECT payload')) return Promise.resolve(cacheRow ? [{ payload: cacheRow }] : [])
      return Promise.resolve([])   // INSERT/upsert — ignore
    },
    { json: (v: object) => v },
  ) as unknown as DbSql
  return fn
}

const server = setupServer()
beforeAll(() => server.listen())
afterEach(() => {
  server.resetHandlers()
  // Reset JWKS singleton between tests by re-assigning the module-level variable
  // Use dynamic import to access the module internals reset helper (see below)
})
afterAll(() => server.close())

describe('verifyBotFrameworkJwt', () => {
  beforeEach(() => {
    // Reset JWKS singleton so each test starts fresh
    vi.resetModules()
  })

  test('valid JWT passes verification and returns payload', async () => {
    const jwks = await buildJwks()
    server.use(
      http.get('https://login.botframework.com/v1/.well-known/keys', () =>
        HttpResponse.json(jwks),
      ),
    )
    const { verifyBotFrameworkJwt: verify } = await import('./jwt.js')
    const token = await signTestJwt()
    const payload = await verify(token, BOT_ID, makeSqlStub())
    expect(payload.aud).toBe(BOT_ID)
    expect(payload.iss).toBe('https://api.botframework.com')
  })

  test('expired token throws BotFrameworkJwtInvalid', async () => {
    const jwks = await buildJwks()
    server.use(
      http.get('https://login.botframework.com/v1/.well-known/keys', () =>
        HttpResponse.json(jwks),
      ),
    )
    const { verifyBotFrameworkJwt: verify } = await import('./jwt.js')
    const token = await new jose.SignJWT({ aud: BOT_ID })
      .setProtectedHeader({ alg: 'RS256', kid: 'test-key-1' })
      .setIssuer('https://api.botframework.com')
      .setIssuedAt(Math.floor(Date.now() / 1000) - 7200)
      .setExpirationTime(Math.floor(Date.now() / 1000) - 3600)
      .sign(privateKey)
    await expect(verify(token, BOT_ID, makeSqlStub())).rejects.toThrow(BotFrameworkJwtInvalid)
  })

  test('wrong audience throws BotFrameworkJwtInvalid', async () => {
    const jwks = await buildJwks()
    server.use(
      http.get('https://login.botframework.com/v1/.well-known/keys', () =>
        HttpResponse.json(jwks),
      ),
    )
    const { verifyBotFrameworkJwt: verify } = await import('./jwt.js')
    const token = await signTestJwt({ aud: 'wrong-bot-id' })
    await expect(verify(token, BOT_ID, makeSqlStub())).rejects.toThrow(BotFrameworkJwtInvalid)
  })

  test('wrong issuer throws BotFrameworkJwtInvalid', async () => {
    const jwks = await buildJwks()
    server.use(
      http.get('https://login.botframework.com/v1/.well-known/keys', () =>
        HttpResponse.json(jwks),
      ),
    )
    const { verifyBotFrameworkJwt: verify } = await import('./jwt.js')
    const token = await new jose.SignJWT({ aud: BOT_ID })
      .setProtectedHeader({ alg: 'RS256', kid: 'test-key-1' })
      .setIssuer('https://evil.example.com')
      .setIssuedAt()
      .setExpirationTime('1h')
      .sign(privateKey)
    await expect(verify(token, BOT_ID, makeSqlStub())).rejects.toThrow(BotFrameworkJwtInvalid)
  })

  test('warm-start: reads JWKS from DB cache, skips remote fetch', async () => {
    // We pass a pre-populated cache row — jose should NOT hit the JWKS URL
    const jwks = await buildJwks()
    const jwksStr = JSON.stringify(jwks)
    // Build a minimal JWKSCacheInput with uat so jose considers it fresh
    const cacheRow = { jwks: jwks.keys, uat: Date.now() + 3_600_000 } as jose.JWKSCacheInput

    let remoteHits = 0
    server.use(
      http.get('https://login.botframework.com/v1/.well-known/keys', () => {
        remoteHits++
        return HttpResponse.json(jwks)
      }),
    )

    const { verifyBotFrameworkJwt: verify } = await import('./jwt.js')
    const token = await signTestJwt()
    await verify(token, BOT_ID, makeSqlStub(cacheRow))
    // If the cache was used, the remote endpoint should not have been fetched
    // (jose uses the pre-seeded cache when uat is in the future)
    expect(remoteHits).toBe(0)
  })
})
```
