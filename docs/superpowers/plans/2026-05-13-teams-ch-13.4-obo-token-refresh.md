# Teams OBO Token Refresh — Implementation Plan (WBS 13.4)

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** When Teams completes SSO sign-in for a user, exchange the Teams SSO token for a Graph OBO token and store it in `oauth.oauth_tokens`. After this task, the token vault is populated so 13.5 (the real Planner handler) can make Graph calls without additional auth ceremony.

**Architecture:** A new `sso.ts` module in `@seta/ms-teams` handles `signin/tokenExchange` invoke activities. `routes.ts` intercepts them before the handler and calls `sso.ts`. `TeamsRouterOpts` gains `provider: OAuthProvider` and `vault: TokenVault` (both already constructed in `apps/api/src/main.ts`). The Teams app manifest already has a `webApplicationInfo` block but uses a hardcoded GUID — it must be corrected to use `{{MS_BOT_ID}}` and gain a `resource` field. Without `resource`, Teams never sends `signin/tokenExchange`.

**Tech Stack:** `@seta/oauth` (`OAuthProvider`, `TokenVault`), `msw` 2.7.x, vitest.

**Spec:** `docs/superpowers/specs/2026-05-13-teams-ch-13.4-obo-token-refresh-design.md`

**Depends on:** WBS 13.3 merged.

## Background: Teams SSO + OBO flow

```
1. User sends any message in Teams 1:1
2. Teams intercepts — user not yet authenticated
3. Teams sends: invoke { name: 'signin/tokenExchange', value: { token: <AAD SSO token> } }
4. sso.ts: provider.acquireOnBehalfOf({ tenantId, userAssertion: ssoToken, scopes })
5. MSAL exchanges SSO token → OBO access token + refresh token
6. vault.put(tenantId, 'entra', `user:${homeAccountId}`, bundle)
7. Bot returns: { type: 'invokeResponse', value: { status: 200 } }
8. Teams re-sends the original user message → mock handler responds as before
```

If admin consent has not been granted yet, step 4 throws and the bot returns `{ status: 412 }` — Teams then shows a "Sign in" button.

### Task 2: Create `sso.ts`

**Files:** Create `modules/channels/teams/src/sso.ts`

- [ ] **Write the file**

```ts
import type { OAuthProvider, TokenVault } from '@seta/oauth'
import { logger } from '@seta/observability'
import type { Activity, OutboundActivity } from './handler.js'

const PLANNER_SCOPES = [
  'https://graph.microsoft.com/Tasks.ReadWrite',
  'https://graph.microsoft.com/User.Read',
]

export async function handleSsoTokenExchange(
  activity: Activity,
  tenantId: string,
  deps: { provider: OAuthProvider; vault: TokenVault },
): Promise<OutboundActivity> {
  const value = activity.value as Record<string, unknown> | undefined
  const userAssertion = typeof value?.token === 'string' ? value.token : undefined

  if (!userAssertion) {
    logger.warn({ conversationId: activity.conversation.id }, 'signin/tokenExchange missing token')
    return { type: 'invokeResponse', value: { status: 400 } }
  }

  try {
    const bundle = await deps.provider.acquireOnBehalfOf({
      tenantId,
      userAssertion,
      scopes: PLANNER_SCOPES,
    })

    const homeAccountId = bundle.meta.homeAccountId as string
    await deps.vault.put(tenantId, 'entra', `user:${homeAccountId}`, bundle)

    return { type: 'invokeResponse', value: { status: 200 } }
  } catch (err) {
    // Consent not yet granted — Teams shows the "Sign in" prompt on 412
    logger.warn({ err, tenantId }, 'OBO exchange failed — likely missing admin consent')
    return { type: 'invokeResponse', value: { status: 412 } }
  }
}
```

```ts
import { describe, expect, test } from 'vitest'
import { handleSsoTokenExchange } from './sso.js'
import type { Activity } from './activity.js'
import type { OAuthProvider, TokenBundle, TokenVault } from '@seta/oauth'

function makeActivity(tokenValue?: string): Activity {
  return {
    type: 'invoke',
    name: 'signin/tokenExchange',
    serviceUrl: 'https://smba.trafficmanager.net/apis',
    channelId: 'msteams',
    from: { id: 'user-1', aadObjectId: 'aad-1' },
    conversation: { id: 'conv-1', conversationType: 'personal' },
    recipient: { id: 'bot-1' },
    value: tokenValue !== undefined ? { token: tokenValue } : {},
  } as Activity
}

function makeBundle(homeAccountId: string): TokenBundle {
  return {
    accessToken: 'access-token',
    refreshToken: 'refresh-token',
    scopes: ['https://graph.microsoft.com/Tasks.ReadWrite'],
    expiresAt: new Date(Date.now() + 3600 * 1000),
    meta: { homeAccountId, tenantId: 'tenant-1' },
  } as unknown as TokenBundle
}

describe('handleSsoTokenExchange', () => {
  test('valid SSO token → stores bundle and returns 200', async () => {
    const bundle = makeBundle('home-account-id-1')
    let storedKey: string | undefined
    let storedBundle: TokenBundle | undefined

    const provider = {
      acquireOnBehalfOf: async () => bundle,
    } as unknown as OAuthProvider

    const vault = {
      put: async (_t: string, _p: string, key: string, b: TokenBundle) => {
        storedKey = key
        storedBundle = b
      },
    } as unknown as TokenVault

    const result = await handleSsoTokenExchange(makeActivity('valid-sso-token'), 'tenant-1', {
      provider,
      vault,
    })

    expect(result).toEqual({ type: 'invokeResponse', value: { status: 200 } })
    expect(storedKey).toBe('user:home-account-id-1')
    expect(storedBundle).toBe(bundle)
  })

  test('OBO fails (no consent) → returns 412, nothing stored', async () => {
    let putCalled = false
    const provider = {
      acquireOnBehalfOf: async () => { throw new Error('invalid_grant') },
    } as unknown as OAuthProvider
    const vault = {
      put: async () => { putCalled = true },
    } as unknown as TokenVault

    const result = await handleSsoTokenExchange(makeActivity('bad-token'), 'tenant-1', {
      provider,
      vault,
    })

    expect(result).toEqual({ type: 'invokeResponse', value: { status: 412 } })
    expect(putCalled).toBe(false)
  })

  test('missing token in value → returns 400', async () => {
    let acquireCalled = false
    const provider = {
      acquireOnBehalfOf: async () => { acquireCalled = true; return {} as TokenBundle },
    } as unknown as OAuthProvider
    const vault = { put: async () => {} } as unknown as TokenVault

    const result = await handleSsoTokenExchange(makeActivity(undefined), 'tenant-1', {
      provider,
      vault,
    })

    expect(result).toEqual({ type: 'invokeResponse', value: { status: 400 } })
    expect(acquireCalled).toBe(false)
  })
})
```

### Task 8: Write and run the integration test

**Files:** Create `modules/channels/teams/tests/integration/sso-token-exchange.test.ts`

- [ ] **Write the integration test**

```ts
import { afterAll, afterEach, beforeAll, expect, test } from 'vitest'
import type { OAuthProvider, TokenBundle } from '@seta/oauth'

const DATABASE_URL = process.env.DATABASE_URL
if (!DATABASE_URL) throw new Error('DATABASE_URL required for integration tests')

const postgres = await import('postgres')
const sql = postgres.default(DATABASE_URL)

const TEST_TENANT = 'integration-tenant-sso-test'
const TEST_ACCOUNT = 'home-account-int-1'

// Use real vault backed by the test DB
const { createTokenVault, createKmsClient } = await import('@seta/oauth')
const kms = createKmsClient({ KMS_PROVIDER: 'env', DEV_DEK_BASE64: Buffer.alloc(32).toString('base64') })
const vault = createTokenVault({ sql: sql as never, kms })

function makeBundle(): TokenBundle {
  return {
    accessToken:  'at-' + Date.now(),
    refreshToken: 'rt-' + Date.now(),
    scopes:       ['https://graph.microsoft.com/Tasks.ReadWrite'],
    expiresAt:    new Date(Date.now() + 3600 * 1000),
    meta:         { homeAccountId: TEST_ACCOUNT, tenantId: TEST_TENANT },
  } as unknown as TokenBundle
}

function makeActivity(token: string) {
  return {
    type: 'invoke',
    name: 'signin/tokenExchange',
    serviceUrl: 'https://smba.trafficmanager.net/apis',
    channelId: 'msteams',
    from: { id: 'user-1', aadObjectId: 'aad-1' },
    conversation: { id: 'conv-1', conversationType: 'personal' },
    recipient: { id: 'bot-1' },
    value: { token },
  }
}

afterAll(async () => {
  await sql`DELETE FROM oauth.oauth_tokens WHERE tenant_id = ${TEST_TENANT}`
  await sql.end()
})

test('signin/tokenExchange writes token bundle to oauth.oauth_tokens', async () => {
  const { handleSsoTokenExchange } = await import('../../src/sso.js')
  const bundle = makeBundle()
  const provider = { acquireOnBehalfOf: async () => bundle } as unknown as OAuthProvider

  const result = await handleSsoTokenExchange(makeActivity('sso-test-token') as never, TEST_TENANT, {
    provider,
    vault,
  })

  expect(result).toEqual({ type: 'invokeResponse', value: { status: 200 } })

  const rows = await sql<{ partition_key: string }[]>`
    SELECT partition_key FROM oauth.oauth_tokens
    WHERE tenant_id = ${TEST_TENANT}
      AND provider_id = 'entra'
      AND partition_key = ${'user:' + TEST_ACCOUNT}
  `
  expect(rows).toHaveLength(1)
})

test('replay of same signin/tokenExchange is idempotent (no duplicate row)', async () => {
  const { handleSsoTokenExchange } = await import('../../src/sso.js')
  const provider = { acquireOnBehalfOf: async () => makeBundle() } as unknown as OAuthProvider

  await handleSsoTokenExchange(makeActivity('sso-token-2') as never, TEST_TENANT, { provider, vault })
  await handleSsoTokenExchange(makeActivity('sso-token-2') as never, TEST_TENANT, { provider, vault })

  const rows = await sql<{ partition_key: string }[]>`
    SELECT partition_key FROM oauth.oauth_tokens
    WHERE tenant_id = ${TEST_TENANT}
      AND provider_id = 'entra'
      AND partition_key = ${'user:' + TEST_ACCOUNT}
  `
  expect(rows).toHaveLength(1)
})
```
