## Business Context

These are the only write operations on the `work_time` table. Create and Update both cascade timesheet recalculation (BR-09) via the `WorkTimeConfigChanged` domain event. The attendance module's event handler consumes this event and recalculates affected rows — work-time must not know about timesheet internals.

The legacy service had two critical bugs that must be fixed here:

1. `forEach` + `async` — cascade recalculation silently never awaited (fixed: use domain event)
2. Empty catch in `updateWorkingTimeService` — errors swallowed (fixed: let errors propagate)

## Drizzle Repository Implementation

```typescript
// infrastructure/repositories/drizzle-work-time.repository.ts
import { Injectable } from '@nestjs/common'
import { InjectDrizzle } from '@future/db'
import { and, eq, isNull, desc, or, sql } from 'drizzle-orm'
import type { NodePgDatabase } from 'drizzle-orm/node-postgres'
import { workTimeTable } from '../schema/time.schema'
import type { IWorkTimeRepository } from '../../domain/repositories/work-time.repository'
import type { WorkTime } from '../../domain/entities/work-time.entity'

@Injectable()
export class DrizzleWorkTimeRepository implements IWorkTimeRepository {
  constructor(@InjectDrizzle() private readonly db: NodePgDatabase) {}

  async findByUserId(userId: string, tenantId: string): Promise<WorkTime[]> {
    return this.db
      .select()
      .from(workTimeTable)
      .where(and(eq(workTimeTable.userId, userId), eq(workTimeTable.tenantId, tenantId)))
      .orderBy(desc(workTimeTable.fromDate))
  }

  async findById(id: string, tenantId: string): Promise<WorkTime | null> {
    const rows = await this.db
      .select()
      .from(workTimeTable)
      .where(and(eq(workTimeTable.id, id), eq(workTimeTable.tenantId, tenantId)))
      .limit(1)
    return rows[0] ?? null
  }

  async findLatestOpenEnded(userId: string, tenantId: string): Promise<WorkTime | null> {
    const rows = await this.db
      .select()
      .from(workTimeTable)
      .where(
        and(
          eq(workTimeTable.userId, userId),
          eq(workTimeTable.tenantId, tenantId),
          isNull(workTimeTable.toDate),
        ),
      )
      .orderBy(desc(workTimeTable.fromDate))
      .limit(1)
    return rows[0] ?? null
  }

  async findOverlapping(
    userId: string,
    tenantId: string,
    fromDate: string,
    toDate: string | null,
    excludeId?: string,
  ): Promise<WorkTime | null> {
    // Build the query range — if toDate is null, use fromDate + 1 day as the range end
    // Uses PostgreSQL daterange overlap operator &&
    const queryRange = toDate
      ? sql`daterange(${fromDate}::date, ${toDate}::date, '[]')`
      : sql`daterange(${fromDate}::date, (${fromDate}::date + interval '1 day')::date, '[]')`

    const existingRange = sql`
      CASE
        WHEN ${workTimeTable.toDate} IS NOT NULL
          THEN daterange(${workTimeTable.fromDate}::date, ${workTimeTable.toDate}::date, '[]')
        ELSE daterange(${workTimeTable.fromDate}::date, (${workTimeTable.fromDate}::date + interval '1 day')::date, '[]')
      END
    `

    const conditions = [
      eq(workTimeTable.userId, userId),
      eq(workTimeTable.tenantId, tenantId),
      sql`${existingRange} && ${queryRange}`,
    ]
    if (excludeId) {
      conditions.push(sql`${workTimeTable.id} != ${excludeId}`)
    }

    const rows = await this.db
      .select()
      .from(workTimeTable)
      .where(and(...conditions))
      .limit(1)
    return rows[0] ?? null
  }

  async insert(data: Omit<WorkTime, 'id'>): Promise<WorkTime> {
    const rows = await this.db.insert(workTimeTable).values(data).returning()
    return rows[0]
  }

  async update(
    id: string,
    tenantId: string,
    data: Partial<
      Pick<
        WorkTime,
        | 'fromDate'
        | 'toDate'
        | 'fromTime'
        | 'toTime'
        | 'startBreakTime'
        | 'endBreakTime'
        | 'description'
      >
    >,
  ): Promise<WorkTime> {
    const rows = await this.db
      .update(workTimeTable)
      .set({ ...data, updatedAt: new Date() })
      .where(and(eq(workTimeTable.id, id), eq(workTimeTable.tenantId, tenantId)))
      .returning()
    if (!rows[0]) throw new Error(`WorkTime ${id} not found`)
    return rows[0]
  }

  async setToDate(id: string, tenantId: string, toDate: string): Promise<void> {
    await this.db
      .update(workTimeTable)
      .set({ toDate, updatedAt: new Date() })
      .where(and(eq(workTimeTable.id, id), eq(workTimeTable.tenantId, tenantId)))
  }

  async delete(id: string, tenantId: string): Promise<void> {
    const result = await this.db
      .delete(workTimeTable)
      .where(and(eq(workTimeTable.id, id), eq(workTimeTable.tenantId, tenantId)))
      .returning({ id: workTimeTable.id })
    if (!result[0]) throw new Error(`WorkTime ${id} not found`)
  }
}
```

### CreateWorkTimeHandler

Business logic (mirrors `createWorktimeService` with bugs fixed):

1. Check overlap via `repo.findOverlapping(userId, tenantId, fromDate, toDate ?? null)` — throw `ConflictException` if found
2. Start transaction:
   a. `repo.findLatestOpenEnded(userId, tenantId)` — if found, close it: `repo.setToDate(latestId, tenantId, fromDate - 1 day)`
   b. `repo.insert(...)` — create new row
3. Publish `WorkTimeConfigChanged` event via `KernelAuditFacade` or outbox (see event publishing pattern below)
4. Return new `WorkTime`
